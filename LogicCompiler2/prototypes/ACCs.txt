[BLOCK ACCs#]
TYPE=BLOCK
SYMBOL=ACCs
CATEGORY=SAP
DESCRIPTION=Accumulateur 8 bits (EA, /LA, CLK)
POS=200,200
WIDTH=120
HEIGHT=100

# Ordre des broches : EA /LA CLK A D
INPUTS=EA;LA;CLK;A
OUTPUTS=D
INTERNALS=N;ACC

# LA actif bas : 0 = charge A, 1 = conserve ACC
EQUATIONS=N   = (/LA * A) + (LA * @ACC)

# Bascule 8 bits
EQUATIONS=ACC = (CLK * @N) + (/CLK * @ACC)

# EA : Enable Accu vers le bus (actif haut)
EQUATIONS=D   = (EA * ACC)
